What I'm Building

A full-stack 5G-connected smart traffic analysis system for the TEKNOFEST 2026 competition organized by Turkcell in partnership with TOGG. The system analyzes live roadside camera feeds using AI and dynamically leverages 5G network programmable APIs to optimize analysis quality in real-time.

The system detects vehicles, reads license plates via OCR, estimates vehicle speed using homography-based pixel-to-meter calibration, and analyzes driver behavior (phone usage, fatigue, seatbelt violations) from the exterior camera through windshield analysis. When a TOGG vehicle is detected, the system requests higher network bandwidth via the 5G Quality on Demand API.

The architecture splits workload: a mobile app (Flutter) handles 5G API calls (Number Verification for passwordless auth, QoD for dynamic bandwidth) and displays results, while a PC backend connected via USB tethering runs all heavy AI inference on GPU. Results flow to the phone in real-time via WebSocket.

from ultralytics import YOLO
import easyocr
from collections import Counter

model = YOLO("weights/traffic_yolov8x.pt")
reader = easyocr.Reader(["tr"])

# Temporal voting buffer per tracked vehicle
plate_buffer = {}  # {track_id: [readings...]}

def detect_and_read(frame, track_id):
    """Detect vehicles + OCR plates with temporal voting."""
    results = model(frame, device="cuda:0", half=True)

    for det in results[0].boxes:
        if det.cls == 1:  # License plate class
            x1, y1, x2, y2 = det.xyxy[0]
            crop = frame[y1:y2, x1:x2]

            # Enhancement: CLAHE + threshold
            enhanced = enhance_plate(crop)

            # OCR reading
            text = reader.readtext(enhanced, detail=0)
            if text and validate_turkish_plate(text[0]):
                plate_buffer[track_id].append(text[0])

    # Majority vote after 30 frames
    if len(plate_buffer[track_id]) >= 30:
        counter = Counter(plate_buffer[track_id])
        return counter.most_common(1)[0][0]
speed_estimation.py
import cv2
import numpy as np

# Homography: perspective ‚Üí bird's eye view
# 4 road reference points (calibrated at setup)
src_pts = np.float32([[p1], [p2], [p3], [p4]])
dst_pts = np.float32([[q1], [q2], [q3], [q4]])
H = cv2.getPerspectiveTransform(src_pts, dst_pts)

# pixels_per_meter from calibration
PPM = 100.0  # 100 pixels = 1 meter in bird's eye

def estimate_speed(pos_prev, pos_curr, fps=30):
    """Convert pixel displacement to km/h."""

    # Transform to bird's eye coordinates
    p1_bev = cv2.perspectiveTransform(
        pos_prev.reshape(1,1,2), H)[0][0]
    p2_bev = cv2.perspectiveTransform(
        pos_curr.reshape(1,1,2), H)[0][0]

    # Pixel distance ‚Üí meters
    dist_px = np.linalg.norm(p2_bev - p1_bev)
    dist_m = dist_px / PPM

    # Time between frames
    dt = 1.0 / fps

    # Speed in km/h
    speed_ms = dist_m / dt
    speed_kmh = speed_ms * 3.6
    return speed_kmh
driver_analysis.py
import mediapipe as mp

mp_face = mp.solutions.face_mesh
face_mesh = mp_face.FaceMesh(
    max_num_faces=1,
    min_detection_confidence=0.5
)

# Eye landmarks for EAR calculation
LEFT_EYE = [362, 385, 387, 263, 373, 380]
RIGHT_EYE = [33, 160, 158, 133, 153, 144]

def eye_aspect_ratio(landmarks, eye_indices):
    """EAR = (|p2-p6| + |p3-p5|) / (2 * |p1-p4|)"""
    pts = [landmarks[i] for i in eye_indices]
    vertical_1 = dist(pts[1], pts[5])
    vertical_2 = dist(pts[2], pts[4])
    horizontal = dist(pts[0], pts[3])
    return (vertical_1 + vertical_2) / (2.0 * horizontal)

def detect_fatigue(windshield_crop):
    """PERCLOS: % of time eyes are closed."""
    results = face_mesh.process(windshield_crop)
    if not results.multi_face_landmarks:
        return None

    lm = results.multi_face_landmarks[0].landmark
    ear_left = eye_aspect_ratio(lm, LEFT_EYE)
    ear_right = eye_aspect_ratio(lm, RIGHT_EYE)
    ear_avg = (ear_left + ear_right) / 2.0

    # EAR < 0.2 for extended time ‚Üí drowsy
    return {"ear": ear_avg, "drowsy": ear_avg < 0.2}
qod_manager.py
import requests

# 5G CAMARA Quality on Demand API
QOD_ENDPOINT = "https://api.5g-camara.turkcell.com/qod/v0"

class QoDManager:
    LEVELS = {
        0: "QOS_E",     # Best effort (480p)
        1: "QOS_S",     # Medium (720p)
        2: "QOS_L",     # High priority (1080p+)
    }

    def set_level(self, level, phone_ip, duration=60):
        """Request QoD from 5G network."""
        payload = {
            "ueId": {"ipv4Addr": phone_ip},
            "qos": self.LEVELS[level],
            "duration": duration
        }
        resp = requests.post(
            f"{QOD_ENDPOINT}/sessions",
            json=payload, headers=self.headers
        )
        return resp.json()

# Trigger logic
def decide_qod(detections, qod):
    if has_togg(detections):
        qod.set_level(2)   # Full quality
    elif has_risky_driver(detections):
        qod.set_level(2)   # Emergency detail
    elif vehicle_approaching(detections):
        qod.set_level(1)   # Start plate reading
    else:
        qod.set_level(0)   # Conserve resources
